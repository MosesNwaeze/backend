/************************************************************************************
** The Feed Entity has articleId as foreign key from articles table ****************
** and gifId foreign key from Gifs table. Its primary key is feedId with type int ****
**************************************************************************************/
class Feed {
	constructor() {
		this.feedId,this.articleId,this.gifId;
	}
 
	// feedId
	getFeedId() {
		return this.feedId;
	}
	setFeedId(id){
		this.feedId = id;
	}
 
	//articleId
	getArticleId(){
		return this.articleId;
	}
	setArticleId(articleId) {
		this.articleId = articleId;
	}
 
	//gifId
	getGifId() {
		return this.gifId;
	}
	setGifId(gifId) {
		this.gifId = gifId;
	}
}
module.exports = Feed;
